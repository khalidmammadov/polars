18/// The physical datatype backing a categorical / enum.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
20pub enum CategoricalPhysical {
21    U8,
22    U16,
23    U32,
24}
25
26impl CategoricalPhysical {
27    pub fn dtype(&self) -> DataType {
28        match self {
29            CategoricalPhysical::U8 => DataType::UInt8,
30            CategoricalPhysical::U16 => DataType::UInt16,
31            CategoricalPhysical::U32 => DataType::UInt32,
32        }
33    }
34
35    pub fn max_categories(&self) -> usize {
36        // We might use T::MAX as an indicator, so the maximum number of categories is T::MAX
37        // (giving T::MAX - 1 as the largest category).
38        match self {
39            CategoricalPhysical::U8 => u8::MAX as usize,
40            CategoricalPhysical::U16 => u16::MAX as usize,
41            CategoricalPhysical::U32 => u32::MAX as usize,
42        }
43    }
44}

46// Used to maintain a 1:1 mapping between Categories' UUID and the Categories objects themselves.
47// This is important for serialization.
48static CATEGORIES_REGISTRY: LazyLock<Mutex<PlHashMap<Uuid, Weak<Categories>>>> =
49    LazyLock::new(|| Mutex::new(PlHashMap::new()));
50
51// Used to make FrozenCategories unique based on their content. This allows comparison of datatypes
52// in constant time by comparing pointers.
53#[expect(clippy::type_complexity)]
54static FROZEN_CATEGORIES_REGISTRY: LazyLock<Mutex<HashTable<(u64, Weak<FrozenCategories>)>>> =
55    LazyLock::new(|| Mutex::new(HashTable::new()));
56
57static FROZEN_CATEGORIES_HASHER: LazyLock<PlSeedableRandomStateQuality> =
58    LazyLock::new(PlSeedableRandomStateQuality::random);
59
60static GLOBAL_CATEGORIES: LazyLock<Arc<Categories>> = LazyLock::new(|| {
61    let categories = Arc::new(Categories {
62        name: PlSmallStr::from_static("__POLARS_GLOBAL_CATEGORIES"),
63        physical: CategoricalPhysical::U32,
64        uuid: Uuid::nil(),
65        mapping: MaybeGcMapping::Gc(Mutex::new(Weak::new())),
66    });
67    CATEGORIES_REGISTRY
68        .lock()
69        .unwrap()
70        .insert(Uuid::nil(), Arc::downgrade(&categories));
71    categories
72});

74/// A (named) object which is used to indicate which categorical data types
75/// have the same mapping. The underlying mapping is dynamic, and if gc is true
76/// may be automatically cleared when the last reference to it goes away.
77pub struct Categories {
78    name: PlSmallStr,
79    physical: CategoricalPhysical,
80    uuid: Uuid,
81    mapping: MaybeGcMapping,
82}
83
84enum MaybeGcMapping {
85    Gc(Mutex<Weak<CategoricalMapping>>),
86    Persistent(Arc<CategoricalMapping>),
87}
88
89impl Categories {
90    /// Creates a new Categories object with the given name and physical type.
91    ///
92    /// If gc is true the underlying categories will automatically get cleaned
93    /// up when the last CategoricalMapping reference goes away, otherwise they
94    /// are persistent.
95    pub fn new(name: PlSmallStr, physical: CategoricalPhysical, gc: bool) -> Arc<Self> {
96        Self::new_with_registry(name, physical, gc, &mut CATEGORIES_REGISTRY.lock().unwrap())
97    }
98
99    /// Returns the Categories object with the given UUID. If the UUID is unknown a new one is created.
100    pub fn from_uuid(
101        name: PlSmallStr,
102        physical: CategoricalPhysical,
103        gc: bool,
104        uuid: Uuid,
105    ) -> Arc<Self> {
106        if uuid.is_nil() {
107            return Self::global();
108        }
109
110        let mut registry = CATEGORIES_REGISTRY.lock().unwrap();
111        if let Some(cats_ref) = registry.get(&uuid) {
112            if let Some(cats) = cats_ref.upgrade() {
113                assert!(
114                    cats.name == name,
115                    "UUID already exists with a different name"
116                );
117                assert!(
118                    cats.physical == physical,
119                    "UUID already exists with a different physical type"
120                );
121                return cats;
122            }
123        }
124        Self::new_with_registry(name, physical, gc, &mut registry)
125    }
126
127    /// Returns the global Categories.
128    pub fn global() -> Arc<Self> {
129        GLOBAL_CATEGORIES.clone()
130    }
131
132    fn new_with_registry(
133        name: PlSmallStr,
134        physical: CategoricalPhysical,
135        gc: bool,
136        registry: &mut PlHashMap<Uuid, Weak<Categories>>,
137    ) -> Arc<Categories> {
138        let uuid = Uuid::new_v4();
139
140        let mapping = if gc {
141            MaybeGcMapping::Gc(Mutex::new(Weak::new()))
142        } else {
143            MaybeGcMapping::Persistent(Arc::new(CategoricalMapping::new(physical.max_categories())))
144        };
145
146        let slf = Arc::new(Self {
147            name,
148            physical,
149            uuid,
150            mapping,
151        });
152        registry.insert(uuid, Arc::downgrade(&slf));
153        slf
154    }
155
156    /// The name of this Categories object (not unique).
157    pub fn name(&self) -> &PlSmallStr {
158        &self.name
159    }
160
161    /// The mapping for this Categories object. If no mapping currently exists
162    /// it creates a new empty mapping.
163    pub fn mapping(&self) -> Arc<CategoricalMapping> {
164        match &self.mapping {
165            MaybeGcMapping::Gc(weak) => {
166                let mut guard = weak.lock().unwrap();
167                if let Some(arc) = guard.upgrade() {
168                    return arc;
169                }
170                let arc = Arc::new(CategoricalMapping::new(self.physical.max_categories()));
171                *guard = Arc::downgrade(&arc);
172                arc
173            },
174            MaybeGcMapping::Persistent(arc) => arc.clone(),
175        }
176    }
177
178    pub fn freeze(&self, physical: CategoricalPhysical) -> Arc<FrozenCategories> {
179        let mapping = self.mapping();
180        let n = mapping.num_cats_upper_bound();
181        FrozenCategories::new(physical, (0..n).flat_map(|i| mapping.cat_to_str(i as u32))).unwrap()
182    }
183}
184
185impl Drop for Categories {
186    fn drop(&mut self) {
187        CATEGORIES_REGISTRY.lock().unwrap().remove(&self.uuid);
188    }
189}

191/// An ordered collection of unique strings with an associated pre-computed
192/// mapping to go from string <-> index.
193///
194/// FrozenCategories are globally unique to facilitate constant-time comparison.
195pub struct FrozenCategories {
196    physical: CategoricalPhysical,
197    combined_hash: u64,
198    categories: Utf8ViewArray,
199    mapping: Arc<CategoricalMapping>,
200}
201
202impl FrozenCategories {
203    /// Creates a new FrozenCategories object (or returns a reference to an existing one
204    /// in case these are already known). Returns an error if the categories are not unique.
205    /// It is guaranteed that the nth string ends up with category n (0-indexed).
206    pub fn new<'a, I: Iterator<Item = &'a str>>(
207        physical: CategoricalPhysical,
208        strings: I,
209    ) -> PolarsResult<Arc<Self>> {
210        let hasher = *FROZEN_CATEGORIES_HASHER;
211        let mut mapping = CategoricalMapping::with_hasher(physical.max_categories(), hasher);
212        let mut builder = Utf8ViewArrayBuilder::new(ArrowDataType::Utf8);
213        builder.reserve(strings.size_hint().0);
214
215        let hasher = PlFixedStateQuality::default();
216        let mut combined_hasher = hasher.build_hasher();
217        for s in strings {
218            let hash = hasher.hash_one(s);
219            combined_hasher.write_u64(hash);
220            mapping.insert_cat_with_hash(s, hash)?;
221            builder.push_value_ignore_validity(s);
222            polars_ensure!(mapping.len() == builder.len(), ComputeError: "FrozenCategories must contain unique strings; found duplicate '{s}'");
223        }
224
225        let combined_hash = combined_hasher.finish();
226        let categories = builder.freeze();
227
228        let mut registry = FROZEN_CATEGORIES_REGISTRY.lock().unwrap();
229        let mut last_compared = None; // We have to store the strong reference to avoid a race condition.
230        match registry.entry(
231            combined_hash,
232            |(hash, weak)| {
233                *hash == combined_hash && {
234                    if let Some(frozen_cats) = weak.upgrade() {
235                        let cmp = frozen_cats.categories == categories
236                            && frozen_cats.physical == physical;
237                        last_compared = Some(frozen_cats);
238                        cmp
239                    } else {
240                        false
241                    }
242                }
243            },
244            |(hash, _weak)| *hash,
245        ) {
246            Entry::Occupied(_) => Ok(last_compared.unwrap()),
247            Entry::Vacant(v) => {
248                let slf = Arc::new(Self {
249                    physical,
250                    combined_hash,
251                    categories,
252                    mapping: Arc::new(mapping),
253                });
254                v.insert((combined_hash, Arc::downgrade(&slf)));
255                Ok(slf)
256            },
257        }
258    }
259
260    /// The mapping for this FrozenCategories object.
261    pub fn mapping(&self) -> &Arc<CategoricalMapping> {
262        &self.mapping
263    }
264}
265
266impl Drop for FrozenCategories {
267    fn drop(&mut self) {
268        let mut registry = FROZEN_CATEGORIES_REGISTRY.lock().unwrap();
269        while let Ok(entry) =
270            registry.find_entry(self.combined_hash, |(_, weak)| weak.strong_count() == 0)
271        {
272            entry.remove();
273        }
274    }
275}
